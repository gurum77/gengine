122 // Represents a simple polygon's edge
123 struct Edge {
124 
125  Point* p, *q;
126 
128  Edge(Point& p1, Point& p2) : p(&p1), q(&p2)
129  {
130  if (p1.y > p2.y) {
131  q = &p1;
132  p = &p2;
133  } else if (p1.y == p2.y) {
134  if (p1.x > p2.x) {
135  q = &p1;
136  p = &p2;
137  } else if (p1.x == p2.x) {
138  // Repeat points
139  assert(false);
140  }
141  }
142 
143  q->edge_list.push_back(this);
144  }
145 };
